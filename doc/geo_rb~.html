def compute_expire_time (params)

  multiplier = {
    'minute' => 1,
    'hour'   => 60,
    'day'    => 60*24,
    'week'   => 60*24*7
  }
  if ! params.has_key?('share_duration_unit')
    raise ArgumentError.new("No share unit")
  end

  if params['share_duration_unit'] == 'manual'
    return
  end

  if ! params.has_key?('share_duration_number')
    raise ArgumentError.new("No share number")
  end

  if ! multiplier.has_key?(params['share_duration_unit'])
    raise ArgumentError.new("No multiplier")
  end
  multiplier[params['share_duration_unit']] * params['share_duration_number']
end

def send_beacon (beacon)
  procname = 'send_beacon_' + params['share_via']
  begin
    (method procname).call(beacon)
  rescue NameError
    error_response "Bad method " + procname
  end
end

def Protocol.error_response (error_msg)
  { error: error_msg, backtrace: caller }
end

def Protocol.process_request_send_position (params)
  # parms: device_id, gps_*
  # returns: OK/ERROR
  begin
    sighting = Sighting.new(device_id:     params['device_id'],
                            gps_longitude: params['gps_longitude'],
                            gps_latitude:  params['gps_latitude'],
                            )
    sighting.save
    {}
  rescue => err
    error_response err.to_s
  end
end

def Protocol.process_request_get_positions (params)
  # params: [device_ids]
  # returns: [{device_id_1, latest gps_*_1, sighting_time_1},
  #           {device_id_2, latest gps_*_2, sighting_time_2}, ...]
  if (! params.has_key?('device_id'))
    return (error_response "No device ID")
  end
  begin
    device_ids_str = params['device_id'].split(',').collect {|did| "'" + did + "'"}.join(',')
    sql = "

    SELECT sightings.* FROM (SELECT id, MAX(updated_at) AS max_updated_at

     FROM     sightings
 GROUP BY device_id
) max_device_id

    JOIN sightings ON sightings.id = max_device_id.id WHERE device_id IN (#{device_ids_str}) “

        {sightings: Sighting.find_by_sql(sql)}
  rescue => err
    error_response err.to_s
  end
end

def Protocol.process_request_get_registration (params)
  if ! params.has_key?('device_id')
    error_response "No device ID"
  else
    device = Device.where("device_id=?", params['device_id']).first
    if defined? device
      device
    else
      # This shouldn't happen.  If there is a device_id, it should be in the DB
      $LOG.error ("No record for "+params['device_id'])
      error_response "Unknown device ID"
    end
  end
end

def Protocol.process_request_register_device (params)
  $LOG.debug (params.inspect)
  $LOG.debug params.has_key?('name')
  if ! params.has_key?('name')
    error_response "No name"
  elsif ! params.has_key?('email')
    error_response "No email"
  elsif ! params.has_key?('device_id')
    error_response "No device ID"
  else
    device = Device.where("device_id=?", params['device_id']).first
    $LOG.debug device
    if ! defined? device
      # This shouldn't happen.  If there is a device_id, it should be in the DB
      $LOG.error ("No record for "+params['device_id'])
      error_response "Unknown device ID"
    else
      device.name = params['name']
      device.email = params['email']
      device.save
      return
    end
  end
end

def Protocol.process_request_share_location (params)
  $LOG.debug params
  begin
    $LOG.debug 'A'
    expire_time = compute_expire_time params
    $LOG.debug expire_time
    device_id = get_device_id params
    $LOG.debug device_id
    if ! params.has_key?("share_via")
      raise ArgumentError.new("No share_via")
    end
    if ! params.has_key?("share_to")
      raise ArgumentError.new("No share_to")
    end
  rescue ArgumentError => e
    $LOG.debug e
    error_response (e.message)
  end
  require 'securerandom'
  share_cred = SecureRandom.base64(20)
  beacon = Beacon.new(expire_time: expire_time,
                      seen_device_id: device_id,
                      share_via: params["share_via"],
                      share_to: params["share_to"],
                      shar_cred: share_cred,
                      )
  send_beacon (beacon)
end

def Protocol.process_request (params)
  $LOG.info (params)
  if params.has_key?('method')
    procname = 'process_request_' + params['method']
    begin
      (method procname).call(params)
    rescue NameError
      error_response "Bad method " + procname
    end
  else
    error_response "No method"
  end
end

def parse_backtrace (backtrace) 
  ar = Array.new
  backtrace.each { |x|
    /(?<path>.*?):(?<line_num>\d+):in `(?<routine>.*)'/ =~ x
    file_base = File.basename(path);
    ar.push({file_base: file_base, line_num: line_num, routine: routine})
  }
  ar
end

def log_error_msg (resp)
  backtrace = parse_backtrace resp[:backtrace]
  backtrace_str = backtrace[0][:file_base] + ':' + backtrace[0][:line_num] + ' ' + backtrace[0][:routine]
  msg = resp[:error] + " at " + backtrace_str
  $LOG.error msg
end

def create_error_html (resp)
  template_file = 'views/error.erb'
  template = File.read(template_file)
  @error = resp[:error]
  @backtrace = parse_backtrace resp[:backtrace]
  @env = ENV.to_hash
  renderer = ERB.new(template)
  renderer.result(binding)
end

def create_device_id (user_agent)
  require 'securerandom'
  device_id = SecureRandom.uuid
  $LOG.info device_id
  $LOG.debug (user_agent)
  begin
    device_record = Device.new(device_id: device_id,
                               user_agent: user_agent)
    device_record.save
  rescue => err
    $LOG.error (err)
  end
  $LOG.debug (device_record)
  device_id
end

def get_device_id (params)
  if (request.cookies['device_id'])
    request.cookies['device_id']
  else
    device_id = create_device_id(request.user_agent)
    response.set_cookie('device_id',
                        { :value   => device_id,
                          :domain  => 'geopeers.com',
                          :expires => Time.new(2038,1,17),
                        })
    device_id
  end
end

get '/api' do
  params['device_id'] = get_device_id params
  params['user_agent'] = request.user_agent
  resp = Protocol.process_request params
  if (resp[:error])
    html = create_error_html (resp)
    halt 500, {'Content-Type' => 'text/html'}, html
  else
    content_type :json
    resp.to_json
  end
end
